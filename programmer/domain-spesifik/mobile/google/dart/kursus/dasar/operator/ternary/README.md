void main() {
  print('\n' * 5);

  var nilai = 50;
  var ucapan = nilai >= 75
      ? 'Operator berhasil di jalankan'
      : nilai; // $ variabel dikembalikan sebagai default
  print('$ucapan\n');

  var value = 10;
  var variabel = value >= 80 ? 'Nilai 80' : 'Nilai 70'; // $ penggunaan seperti if-else
  print(variabel);

  print('\n' * 5);
}
